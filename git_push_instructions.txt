# ============================================
# STEP 1: Initialize Git Repository
# ============================================

# Navigate to project directory (if not already there)
cd "C:\Users\walee\Desktop\LLM Invertible Project"

# Initialize git repository
git init

# Add the remote repository
git remote add origin https://github.com/WaleedUmar007/LLM-Interjectible-Script.git

# ============================================
# STEP 2: Configure Git (if not done before)
# ============================================

# Set your name and email (replace with your info)
git config user.name "WaleedUmar007"
git config user.email "your-email@example.com"

# ============================================
# STEP 3: Add Files to Git
# ============================================

# The .gitignore will automatically exclude unwanted files
# Add all files (gitignore will filter)
git add .

# Check what will be committed
git status

# You should see:
# ✓ README.md
# ✓ compare_encoder_decoder.py
# ✓ Comparison_Results/
# ✓ requirements.txt
# ✓ testingData.json
# ✓ .gitignore

# ============================================
# STEP 4: Commit Changes
# ============================================

git commit -m "Initial commit: Encoder vs Decoder Injectivity Comparison"

# Or with a more detailed message:
git commit -m "Add comparative analysis of GPT-2 and BERT injectivity

- Implementation of encoder vs decoder comparison
- Results showing both architectures are injective
- Complete documentation and analysis
- Based on paper: Language Models are Injective and Hence Invertible"

# ============================================
# STEP 5: Push to GitHub
# ============================================

# Push to main branch
git push -u origin main

# If the above fails (branch might be called 'master'), try:
git push -u origin master

# Or rename branch to main first:
git branch -M main
git push -u origin main

git pull origin main --allow-unrelated-histories
